    function handleCommand(cmd, arg) {
        if (cmd === 'help') {
            printNormal("Available commands:\n- ls: List all main directories\n- cd [folder]: Enter a directory\n- read [ID]: Access a specific file\n- clear: Reset terminal screen");
        } 
        else if (cmd === 'ls') {
            printNormal("Directories found:\n/Drafts\n/Logs\n/Analysis\n/Philosophy");
        } 
        else if (cmd === 'cd') {
            if (!arg) return printWarn("Usage: cd [Folder_Name]");
            const folderName = arg.charAt(0).toUpperCase() + arg.slice(1).toLowerCase();
            const db = Global_DB[folderName];
            if (db) {
                let list = `Directory: /${folderName}\n`;
                db.forEach(f => list += `> [ID: ${f.id}] ${f.title}\n`);
                printNormal(list);
            } else {
                printWarn(`Error: Directory '${arg}' not found.`);
            }
        } 
        else if (cmd === 'read') {
            if (!arg) return printWarn("Usage: read [ID]");
            let entry = null;
            for (let key in Global_DB) {
                entry = Global_DB[key].find(f => f.id === arg.toUpperCase());
                if (entry) break;
            }
            
            if (entry) {
                printBilingual(entry);
            } else {
                printWarn(`Error: File ID '${arg}' not recognized or encrypted.`);
            }
        }

            history.innerHTML = '';
        }

        }
        else {
            printNormal(`Unknown command: ${cmd}`);
        }
    }

    // --- 打字机动效逻辑 ---

    async function typeWriter(element, text, speed) {
        // 将文本按换行符分割，处理真实换行
        const lines = text.split('\n');
        for (let j = 0; j < lines.length; j++) {
            const line = lines[j];
            for (let i = 0; i < line.length; i++) {
                element.innerHTML += line.charAt(i);
                scrollTerminal();
                await new Promise(res => setTimeout(res, speed));
            }
            if (j < lines.length - 1) {
                element.innerHTML += "<br>"; // 英文部分换行
            }
        }
    }
